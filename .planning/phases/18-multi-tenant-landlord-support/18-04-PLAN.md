Create `src/app/[landlord]/layout.tsx`:

```typescript
import { notFound } from "next/navigation"
import { getLandlordBySlug } from "@/lib/landlord"
import { LandlordProvider } from "@/contexts/landlord-context"

export default async function LandlordLayout({
  children,
  params,
}: {
  children: React.ReactNode
  params: Promise<{ landlord: string }>
}) {
  const { landlord: slug } = await params
  const landlord = await getLandlordBySlug(slug)
  if (!landlord) notFound()

  // Compute derived color values for buttons
  // #7c3d18 and #6a3214 are darker shades of accent — derive via CSS color-mix or hardcode derivation
  const cssVars = {
    "--bg-color": landlord.bgColor,
    "--text-color": landlord.textColor,
    "--accent-color": landlord.accentColor,
  } as React.CSSProperties

  return (
    <div style={cssVars}>
      <LandlordProvider
        value={{
          id: landlord.id,
          slug: landlord.slug,
          name: landlord.name,
          ownerName: landlord.ownerName,
          address: landlord.address,
          email: landlord.email,
          phone: landlord.phone,
          bgColor: landlord.bgColor,
          textColor: landlord.textColor,
          accentColor: landlord.accentColor,
        }}
      >
        {children}
      </LandlordProvider>
    </div>
  )
}
```

Note: The wrapper `<div>` with CSS variables will be the ancestor of all landlord-scoped pages. The CSS variables `--bg-color`, `--text-color`, and `--accent-color` will cascade to all child elements, making `var(--bg-color)` available in both inline styles and `<style>` tags.
</action>
