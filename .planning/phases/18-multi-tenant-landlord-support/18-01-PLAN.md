# Plan 18-01: Schema & Migration

## Goal
Add the `Landlord` model to the Prisma schema, wire it to `Room` and `Settings`, create a seed script to backfill the existing "High Hill" data, and create the shared helper functions (`getLandlordForAdmin`, `getLandlordBySlug`) and React context that all subsequent plans depend on.

- [ ] `Landlord` model exists in `prisma/schema.prisma` with all required fields
- [ ] `Room` model has a required `landlordId` field with `@relation` to `Landlord`
- [ ] `Settings` model uses `landlordId` (unique) instead of hardcoded `id: "global"`
- [ ] `prisma/seed.ts` creates the "High Hill" landlord and backfills existing rooms/settings
- [ ] `src/lib/landlord.ts` exports `getLandlordForAdmin`, `requireLandlordForAdmin`, `getLandlordBySlug`

1. Add the `Landlord` model after the `datasource db` block and before the `Room` model:

```prisma
model Landlord {
  id          String    @id @default(cuid())
  slug        String    @unique
  name        String
  ownerName   String
  address     String
  email       String
  phone       String?
  bgColor     String    @default("#3a392a")
  textColor   String    @default("#f0ebe0")
  accentColor String    @default("#d4956a")
  adminUserId String    @unique
  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt
  rooms       Room[]
  settings    Settings?
}
```

2. Add to the `Room` model (after the existing fields, before the relations):

```prisma
  landlordId  String
  landlord    Landlord @relation(fields: [landlordId], references: [id])
```

And add an index: `@@index([landlordId])` inside the Room model.

3. Change the `Settings` model from:

```prisma
model Settings {
  id                String   @id @default("global")
  serviceFeePercent Decimal  @db.Decimal(5, 2)
  depositAmount     Decimal  @db.Decimal(10, 2)
  etransferEmail    String?
  updatedAt         DateTime @updatedAt
}
```

to:

```prisma
model Settings {
  id                String   @id @default(cuid())
  landlordId        String   @unique
  landlord          Landlord @relation(fields: [landlordId], references: [id])
  serviceFeePercent Decimal  @db.Decimal(5, 2)
  depositAmount     Decimal  @db.Decimal(10, 2)
  etransferEmail    String?
  updatedAt         DateTime @updatedAt
}
```

<task id="18-01-02">
<title>Create and run Prisma migration</title>
<read_first>
- `prisma/schema.prisma` — to verify the schema changes from task 01 are in place
</read_first>
<action>
Because the existing database has Room rows with no `landlordId` and a Settings row with `id = "global"`, we need a multi-step migration approach:

**Step 1: Create a custom migration** (do NOT use `prisma migrate dev --create-only` with the final schema since landlordId is required). Instead:

1. Temporarily make `landlordId` optional on `Room` (add `?` to `String`) in the schema
2. Run `npx prisma migrate dev --name add_landlord_model` to generate the migration SQL
3. After the migration SQL is created, edit the migration SQL file to add data backfill steps:
   - After the `CREATE TABLE "Landlord"` and `ALTER TABLE "Room" ADD COLUMN "landlordId"` statements, add:
   ```sql
   -- Backfill: Create the default landlord
   INSERT INTO "Landlord" ("id", "slug", "name", "ownerName", "address", "email", "bgColor", "textColor", "accentColor", "adminUserId", "createdAt", "updatedAt")
   VALUES ('default-landlord', 'highhill', 'Leon''s Home', 'Leon', '9 Highhill Dr, Scarborough, ON', 'highhill@uptrendinvestments.net', '#3a392a', '#f0ebe0', '#d4956a', 'PLACEHOLDER_ADMIN_USER_ID', NOW(), NOW());

   -- Backfill: Assign all existing rooms to the default landlord
   UPDATE "Room" SET "landlordId" = 'default-landlord' WHERE "landlordId" IS NULL;

   -- Backfill: Create new landlord-scoped settings from existing global settings
   INSERT INTO "Settings" ("id", "landlordId", "serviceFeePercent", "depositAmount", "etransferEmail", "updatedAt")
   SELECT 'settings-highhill', 'default-landlord', "serviceFeePercent", "depositAmount", "etransferEmail", "updatedAt"
   FROM "Settings" WHERE "id" = 'global';

   -- Delete old global settings row
   DELETE FROM "Settings" WHERE "id" = 'global';
   ```
4. After the backfill, make `landlordId` non-nullable: `ALTER TABLE "Room" ALTER COLUMN "landlordId" SET NOT NULL;`
5. Remove the `?` from `landlordId` on `Room` in the schema file (make it required again)
6. Run `npx prisma migrate dev` to apply

**Important:** The `PLACEHOLDER_ADMIN_USER_ID` must be replaced with the actual Supabase admin user ID. This should come from the `ADMIN_USER_ID` environment variable. Since raw SQL can't read env vars, the seed script (task 03) will update this value. For the migration to work, use a deterministic placeholder that the seed script will overwrite.

**Alternative (simpler) approach if the migration tool struggles:** Use `npx prisma migrate dev --create-only --name add_landlord_model`, then manually edit the generated SQL, then run `npx prisma migrate dev` to apply.
</action>

Create `prisma/seed.ts`:

```typescript
import { PrismaClient } from "@prisma/client"

const prisma = new PrismaClient()

async function main() {
  const adminUserId = process.env.ADMIN_USER_ID
  if (!adminUserId) {
    throw new Error("ADMIN_USER_ID environment variable is required for seeding")
  }

  // Upsert the High Hill landlord
  const landlord = await prisma.landlord.upsert({
    where: { slug: "highhill" },
    create: {
      slug: "highhill",
      name: "Leon's Home",
      ownerName: "Leon",
      address: "9 Highhill Dr, Scarborough, ON",
      email: process.env.LANDLORD_EMAIL ?? "highhill@uptrendinvestments.net",
      phone: null,
      bgColor: "#3a392a",
      textColor: "#f0ebe0",
      accentColor: "#d4956a",
      adminUserId,
    },
    update: {
      adminUserId,
      email: process.env.LANDLORD_EMAIL ?? "highhill@uptrendinvestments.net",
    },
  })

  console.log(`Landlord seeded: ${landlord.name} (${landlord.slug})`)

  // Backfill any rooms that don't have a landlordId yet
  const updated = await prisma.room.updateMany({
    where: { landlordId: landlord.id },
    data: {},  // no-op — just to count
  })

  // Also assign any orphan rooms (from migration placeholder)
  const orphanRooms = await prisma.room.updateMany({
    where: {
      landlordId: "default-landlord",
    },
    data: {
      landlordId: landlord.id,
    },
  })
  if (orphanRooms.count > 0) {
    console.log(`Reassigned ${orphanRooms.count} rooms from placeholder to ${landlord.id}`)
  }

  // Ensure settings exist for this landlord
  const existingSettings = await prisma.settings.findUnique({
    where: { landlordId: landlord.id },
  })
  if (!existingSettings) {
    // Check for settings with placeholder landlord ID
    const placeholderSettings = await prisma.settings.findFirst({
      where: { landlordId: "default-landlord" },
    })
    if (placeholderSettings) {
      await prisma.settings.update({
        where: { id: placeholderSettings.id },
        data: { landlordId: landlord.id },
      })
      console.log("Migrated placeholder settings to real landlord ID")
    } else {
      await prisma.settings.create({
        data: {
          landlordId: landlord.id,
          serviceFeePercent: 14,
          depositAmount: 100,
        },
      })
      console.log("Created default settings for landlord")
    }
  }

  // Clean up placeholder landlord row if it exists
  try {
    await prisma.landlord.delete({ where: { id: "default-landlord" } })
    console.log("Cleaned up placeholder landlord row")
  } catch {
    // Doesn't exist or has dependencies — that's fine
  }

  console.log("Seed complete.")
}

main()
  .catch((e) => {
    console.error(e)
    process.exit(1)
  })
  .finally(async () => {
    await prisma.$disconnect()
  })
```

```json
}
```

Also ensure `tsx` is a dev dependency: `npm install -D tsx` (if not already present).

```typescript
import { createClient } from "@/lib/supabase/server"
import { prisma } from "@/lib/prisma"
import { redirect } from "next/navigation"

/**
 * For server actions — throws on failure (action will return 500).
 * Returns the full Landlord row for the currently logged-in admin.
 */
export async function getLandlordForAdmin() {
  const supabase = await createClient()
  const { data: { user }, error } = await supabase.auth.getUser()
  if (error || !user) throw new Error("Unauthorized")

  const landlord = await prisma.landlord.findUnique({
    where: { adminUserId: user.id },
  })
  if (!landlord) throw new Error("No landlord found for this admin user")

  return landlord
}

/**
 * For RSC pages — redirects on failure instead of throwing.
 * Returns the full Landlord row for the currently logged-in admin.
 */
export async function requireLandlordForAdmin() {
  const supabase = await createClient()
  const { data: { user }, error } = await supabase.auth.getUser()
  if (error || !user) redirect("/login")

  const landlord = await prisma.landlord.findUnique({
    where: { adminUserId: user.id },
  })
  if (!landlord) redirect("/login?error=no_landlord")

  return landlord
}

/**
 * For guest-facing pages — returns null if slug doesn't match any landlord.
 * Caller should use notFound() if null.
 */
export async function getLandlordBySlug(slug: string) {
  return prisma.landlord.findUnique({ where: { slug } })
}
```

```typescript
"use client"

import { createContext, useContext } from "react"

export type LandlordInfo = {
  id: string
  slug: string
  name: string
  ownerName: string
  address: string
  email: string
  phone: string | null
  bgColor: string
  textColor: string
  accentColor: string
}

const LandlordContext = createContext<LandlordInfo | null>(null)

export function LandlordProvider({
  children,
  value,
}: {
  children: React.ReactNode
  value: LandlordInfo
}) {
  return (
    <LandlordContext.Provider value={value}>
      {children}
    </LandlordContext.Provider>
  )
}

export function useLandlord() {
  const ctx = useContext(LandlordContext)
  if (!ctx) throw new Error("useLandlord must be used within a LandlordProvider")
  return ctx
}
```
